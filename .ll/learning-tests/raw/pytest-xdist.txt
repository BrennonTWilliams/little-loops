"""Minimal proof script for pytest-xdist behavior (BUG-2524 explore-api).

Probes five claims about pytest-xdist 3.8.0 by:
  1) Spawning an actual xdist run (`pytest -n 2`) and observing worker Config attributes
     via a probe plugin loaded from conftest.
  2) Spawning a controller-only run (`pytest -n 0`) for the negative case.
  3) Checking marker registration in `[tool.pytest.ini_options].markers`.
  4) Checking that `item.add_marker(pytest.mark.skip(...))` is honored on a worker.
  5) Exercising `pytest_xdist_auto_num_workers(config)` as a callable hook.

The probe plugin writes its observed `config.workerinput` value to a sentinel file,
which the script then reads and asserts on.
"""

from __future__ import annotations

import json
import os
import shutil
import subprocess
import sys
from pathlib import Path

import pytest  # for claim 4 API probe

ROOT = Path(__file__).parent
PROBE_DIR = ROOT / "_probe"
SENTINEL = PROBE_DIR / "observed.json"


def _write_probe_files() -> None:
    PROBE_DIR.mkdir(parents=True, exist_ok=True)
    SENTINEL.unlink(missing_ok=True)

    # Sentinel test that imports cleanly (no markers, no fixtures — minimal).
    (PROBE_DIR / "test_probe.py").write_text(
        "def test_one():\n"
        "    assert True\n"
    )

    # Probe conftest: pytest_configure writes the worker's observed Config shape.
    (PROBE_DIR / "conftest.py").write_text(
        "import json\n"
        "from pathlib import Path\n"
        "import pytest\n"
        "\n"
        "_SENTINEL = Path(__file__).parent / 'observed.json'\n"
        "\n"
        "def pytest_configure(config):\n"
        "    obs = {\n"
        "        'has_workerinput': hasattr(config, 'workerinput'),\n"
        "        'workerinput_value': getattr(config, 'workerinput', '__MISSING__'),\n"
        "        'workerinput_truthy': bool(getattr(config, 'workerinput', None)),\n"
        "        'is_xdist_controller': getattr(config, 'workerinput', '__MISSING__') == '__MISSING__',\n"
        "        'pytest_xdist_auto_num_workers_callable': callable(getattr(config, 'pytest_xdist_auto_num_workers', None)),\n"
        "    }\n"
        "    with _SENTINEL.open('a') as f:\n"
        "        f.write(json.dumps(obs) + '\\n')\n"
    )


def _run_pytest(*args: str) -> tuple[int, str]:
    proc = subprocess.run(
        ["python", "-m", "pytest", *args, "-q", "--no-header", "-p", "no:cacheprovider"],
        cwd=PROBE_DIR,
        capture_output=True,
        text=True,
        env={**os.environ, "PYTHONPATH": str(PROBE_DIR)},
    )
    return proc.returncode, proc.stdout + "\n" + proc.stderr


def _read_observations() -> list[dict]:
    if not SENTINEL.exists():
        return []
    return [json.loads(line) for line in SENTINEL.read_text().splitlines() if line.strip()]


def main() -> int:
    results: dict[str, str] = {}
    notes: list[str] = []

    _write_probe_files()

    # ---- Claim 5: pytest_xdist_auto_num_workers is callable as a hook ----
    try:
        from xdist.plugin import pytest_xdist_auto_num_workers  # type: ignore[import-not-found]

        # In xdist 3.8.0 the hook lives at xdist.plugin. Some versions have it
        # at xdist.newhooks. Try both before giving up.
    except ImportError:
        try:
            from xdist.newhooks import pytest_xdist_auto_num_workers  # type: ignore[no-redef]
        except ImportError:
            pytest_xdist_auto_num_workers = None

    if pytest_xdist_auto_num_workers is None:
        results["claim_5_callable_hook"] = "fail"
        notes.append("pytest_xdist_auto_num_workers import failed in both xdist.plugin and xdist.newhooks")
    else:
        from unittest.mock import MagicMock
        try:
            ret = pytest_xdist_auto_num_workers(MagicMock())
            results["claim_5_callable_hook"] = (
                "pass" if isinstance(ret, int) else "fail"
            )
            notes.append(f"returned {type(ret).__name__}={ret!r}")
        except Exception as exc:  # pragma: no cover
            results["claim_5_callable_hook"] = "fail"
            notes.append(f"raised {type(exc).__name__}: {exc}")

    # ---- Claim 1 + 2: worker has workerinput, controller does NOT ----
    # Run with -n 2 (xdist). Expect: 1 controller observation (no workerinput)
    # + 2 worker observations (workerinput is {"workerid": "gw0"/"gw1"}).
    rc_xdist, out_xdist = _run_pytest("-n", "2")
    obs_xdist = _read_observations()
    if rc_xdist != 0:
        results["claim_1_worker_has_workerinput"] = "fail"
        results["claim_2_controller_no_workerinput"] = "fail"
        notes.append(f"xdist run failed (rc={rc_xdist}); output:\n{out_xdist[-400:]}")
    else:
        workers = [o for o in obs_xdist if o.get("workerinput_truthy")]
        controllers = [o for o in obs_xdist if o.get("is_xdist_controller")]
        results["claim_1_worker_has_workerinput"] = (
            "pass" if workers and all("workerid" in (w.get("workerinput_value") or {}) for w in workers) else "fail"
        )
        results["claim_2_controller_no_workerinput"] = (
            "pass" if controllers and all(not c.get("has_workerinput", True) for c in controllers) else "fail"
        )
        notes.append(
            f"xdist run: {len(workers)} worker observation(s), {len(controllers)} controller observation(s); "
            f"worker workerinput samples = {[w.get('workerinput_value') for w in workers]}"
        )

    # ---- Claim 3: markers declared in pyproject.toml are accepted by --strict-markers ----
    # Verify by parsing pyproject.toml and confirming 'no_parallel' is registered.
    pyproject = Path("/Users/brennon/AIProjects/brenentech/little-loops/scripts/pyproject.toml")
    pyproject_text = pyproject.read_text() if pyproject.exists() else ""
    results["claim_3_marker_registered"] = (
        "pass" if 'no_parallel' in pyproject_text and 'markers' in pyproject_text else "fail"
    )
    notes.append("checked scripts/pyproject.toml for 'no_parallel' in [tool.pytest.ini_options].markers")

    # ---- Claim 4: pytest.mark.skip(reason=...) constructs a MarkDecorator named "skip" ----
    # Verify the API surface that conftest.py:98 uses: pytest.mark.skip(reason=...).
    # The "honored at runtime" half is proven by scripts/tests/test_conftest_cap.py:154-174
    # (existing project test) — this probe focuses on the API shape itself.
    try:
        skip_marker = pytest.mark.skip(reason="no_parallel: probe")
        results["claim_4_marker_api_shape"] = (
            "pass"
            if hasattr(skip_marker, "mark")
            and skip_marker.mark.name == "skip"
            and skip_marker.mark.kwargs.get("reason") == "no_parallel: probe"
            else "fail"
        )
        notes.append(
            f"pytest.mark.skip(reason=...) → mark.name={getattr(skip_marker.mark, 'name', '?')!r}, "
            f"kwargs={dict(skip_marker.mark.kwargs)!r}"
        )
    except Exception as exc:  # pragma: no cover
        results["claim_4_marker_api_shape"] = "fail"
        notes.append(f"raised {type(exc).__name__}: {exc}")

    # ---- Output ----
    print(json.dumps({"results": results, "notes": notes}, indent=2))
    return 0 if all(v == "pass" for v in results.values()) else 1
